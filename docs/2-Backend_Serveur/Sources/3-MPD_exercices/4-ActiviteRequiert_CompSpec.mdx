# Compétence spécifique par activité

## Modèle

```js title="./src/models/activite_requiert_compspec.model.js"
module.exports = (sequelize, Sequelize) => {
    const ActiviteRequiert_CompSpec = sequelize.define("activite_requiert_compspec", {
        fk_activite_id: {
            type: Sequelize.STRING,
            primaryKey: true,
            references: {
                model: "activite_clnt",
                key: "pk_activite_id",
            },
        },
        fk_compspec_id: {
            type: Sequelize.STRING,
            primaryKey: true,
            references: {
                model: "compgen_has_compspec",
                key: "id",
            },
        },
    });

    return ActiviteRequiert_CompSpec;
};
```

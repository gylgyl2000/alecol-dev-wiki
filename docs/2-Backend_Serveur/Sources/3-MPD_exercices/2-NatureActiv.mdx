## Service

```js title="./src/services/alecol.natureactiv.service.js"
import http from "../http-common";

class NatureActivDataService {
    getAll() {
        return http.get("/natureactiv");
    }

    get(id) {
        return http.get(`/natureactiv/${id}`);
    }

    findByNom(nom) {
        return http.get(`/natureactiv?nom=${nom}`);
    }
}

export default new NatureActivDataService();
```

## Contrôleur

```js title="./src/controllers/alecol.natureactiv.controllers.js"
const db = require("../models");
const NatureActiv = db.natureActiv;
const Op = db.Sequelize.Op;

// Récupérer toutes les natures d'activité de la base de données
exports.findAll = (req, res) => {
    const nom = req.query.nom;

    var nameCondition = nom ? { nom: { [Op.like]: `${nom}` } } : null;

    var condition = nameCondition;

    NatureActiv.findAll({ where: condition })
        .then((data) => {
            res.send(data);
        })
        .catch((err) => {
            res.status(500).send({
                message: err.message || "Une erreur s'est produite lors de la récupération des natures d'activité.",
            });
        });
};

// Trouver une seule nature d'activité avec un identifiant
exports.findOne = (req, res) => {
    const id = req.params.id;

    NatureActiv.findByPk(id)
        .then((data) => {
            res.send(data);
        })
        .catch((err) => {
            res.status(500).send({
                message: "Erreur lors de la récupération de la nature d'activité avec l'id=" + id,
            });
        });
};
```

## Route

```js title="./src/routes/alecol.natureactiv.routes.js"
module.exports = (app) => {
    const natureactiv = require("../controllers/alecol.natureactiv.controllers");

    var router = require("express").Router();

    // Récupérer toutes les natures d'activité
    router.get("/", natureactiv.findAll);

    // Récupérer une seule nature d'activité avec id
    router.get("/:id", natureactiv.findOne);

    app.use("/api/natureactiv", router);
};
```
